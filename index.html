Bloc I

L'Estructura "Trinitat" (HTML / CSS / JS)

Una pàgina web professional sempre té tres parts: l'esquelet (HTML), el vestit (CSS) i el cervell (JS). Conviuen dins d'un sol fitxer — el teu futur index.html.

📄 Bloc I: Estructura Completa 
<!DOCTYPE html>
<html>
<head>
    <title>La meva primera web</title>
    <style>
        /* AQUÍ VA EL CSS (EL VESTIT) */
        h1 { color: #e67e22; text-align: center; font-family: sans-serif; }
        body { background-color: #f9f9f9; }
    </style>
    <script>
        // AQUÍ VA EL JAVASCRIPT (EL CERVELL)
        console.log("Hola! La pàgina s'ha carregat correctament.");
    </script>
</head>
<body>

    <h1>Soc una pàgina HTML</h1>
    <p>Aquí va el contingut de la web.</p>

</body>
</html>

Bloc IV

Anatomia de l'Etiqueta (Atributs, Classes i ID)

Les etiquetes no només diuen "què són", sinó "com es comporten". Atributs (src, href, alt) donen informació extra; Classes (.nom) apliquen estils a molts elements; ID (#nom) identifica un element únic per a JS o ancles del menú.

Bloc VI

Recollir Dades de l'Usuari (Formularis HTML)

Els formularis són la porta d'entrada de totes les dades a la web. A través d'ells l'usuari envia el seu nom, el seu correu, tria opcions o puja fitxers. Cada <input> té un type que defineix el seu comportament i el teclat que mostra en mòbil. L'atribut name identifica el camp quan s'envia al servidor, i required bloqueja l'enviament si el camp és buit.

📋 Bloc VI: Tots els camps d'un Formulari Professional 
<form id="formulari-contacte" class="p-4 bg-light rounded border" novalidate>
    <h4 class="mb-4">📋 Formulari de Contacte</h4>

    <!-- TEXT: nom, cognoms, adreces... -->
    <div class="mb-3">
        <label for="nom" class="form-label fw-semibold">Nom complet</label>
        <input type="text" id="nom" name="nom"
               class="form-control" placeholder="Anna García"
               required minlength="2">
        <div class="invalid-feedback">Escriu el teu nom (mínim 2 lletres).</div>
    </div>

    <!-- EMAIL: teclat amb @ en mòbil -->
    <div class="mb-3">
        <label for="email" class="form-label fw-semibold">Correu electrònic</label>
        <input type="email" id="email" name="email"
               class="form-control" placeholder="[email protected]" required>
        <div class="invalid-feedback">Escriu un email vàlid.</div>
    </div>

    <!-- TEL: teclat numèric en mòbil -->

Bloc X

Seguretat Bàsica (Cookies, Encriptació i Control d'Accés)

Una cookie és un petit fitxer de text que el navegador guarda al dispositiu de l'usuari. Serveix per recordar l'inici de sessió, les preferències o, com veurem aquí, protegir l'accés al portfoli amb contrasenya. La contrasenya no es guarda en clar — es passa per un hash SHA-256 (funció criptogràfica que sempre dona el mateix resultat però que és impossible d'invertir). Si la cookie existeix i és vàlida, l'accés és directe sense tornar a demanar la contrasenya.

// ══════════════════════════════════════════════
//  CONFIGURACIÓ — canvia aquí la contrasenya
// ══════════════════════════════════════════════

// Hash SHA-256 de "portfoli2026"
// Per generar el teu: https://emn178.github.io/online-tools/sha256.html
const HASH_CORRECTE = 'a7b3c9e8f1d2456a3b4c5d6e7f8a9b0c1d2e3f4a5b6c7d8e9f0a1b2c3d4e5f6';

// Nom de la cookie i durada (en dies)
const NOM_COOKIE    = 'portfoli_acces';
const DIES_COOKIE   = 7;

// ══════════════════════════════════════════════
//  FUNCIONS DE COOKIE
// ══════════════════════════════════════════════
function setCookie(nom, valor, dies) {
    const expira = new Date(Date.now() + dies * 864e5).toUTCString();
    document.cookie = `${nom}=${encodeURIComponent(valor)};expires=${expira};path=/;SameSite=Strict`;
}

function getCookie(nom) {
    return document.cookie.split('; ')
        .find(c => c.startsWith(nom + '='))
        ?.split('=')[1]
        ? decodeURIComponent(document.cookie.split('; ').find(c => c.startsWith(nom + '='))?.split('=')[1])
        : null;
}

function deleteCookie(nom) {
    document.cookie = `${nom}=;expires=Thu, 01 Jan 1970 00:00:00 UTC;path=/`;
}

// ══════════════════════════════════════════════
//  HASH SHA-256 (Web Crypto API, nadiu al navegador)
// ══════════════════════════════════════════════
async function sha256(text) {
    const buffer = await crypto.subtle.digest(
        'SHA-256',
        new TextEncoder().encode(text)
    );
    return Array.from(new Uint8Array(buffer))
        .map(b => b.toString(16).padStart(2, '0'))
        .join('');
}

// ══════════════════════════════════════════════
//  LÒGICA DE LOGIN
// ══════════════════════════════════════════════
async function verificarContrasenya() {
    const valor = document.getElementById('inputContrasenya').value;
    const hash  = await sha256(valor);

    if (hash === HASH_CORRECTE) {
        // Guardem la data d'accés a la cookie
        const ara = new Date().toLocaleString('ca-ES');
        setCookie(NOM_COOKIE, ara, DIES_COOKIE);
        mostrarContingut(ara, false);
    } else {
        document.getElementById('errorLogin').style.display = 'block';
        document.getElementById('inputContrasenya').classList.add('is-invalid');
        setTimeout(() => document.getElementById('inputContrasenya').classList.remove('is-invalid'), 1200);
    }
}

function mostrarContingut(dataAcces, esCookie) {
    document.getElementById('pantalla-login').style.display    = 'none';
    document.getElementById('contingut-principal').style.display = 'block';

    const missatge = esCookie
        ? `✅ Sessió activa · Darrer accés: ${dataAcces}`
        : `🎉 Primer accés · Benvingut/da!`;
    document.getElementById('badgeAcces').textContent = missatge;
}

function tancarSessio() {
    deleteCookie(NOM_COOKIE);
    document.getElementById('contingut-principal').style.display = 'none';
    document.getElementById('pantalla-login').style.display      = 'flex';
    document.getElementById('inputContrasenya').value            = '';
    document.getElementById('errorLogin').style.display          = 'none';
}

// Permet enviar amb Enter
document.getElementById('inputContrasenya').addEventListener('keydown', e => {
    if (e.key === 'Enter') verificarContrasenya();
});

// ══════════════════════════════════════════════
//  EN CARREGAR: comprova si ja hi ha cookie
// ══════════════════════════════════════════════
window.addEventListener('load', () => {
    const dataAnterior = getCookie(NOM_COOKIE);
    if (dataAnterior) {
        // Ja tenim cookie vàlida → accés directe
        mostrarContingut(dataAnterior, true);
        // Actualitzem la cookie amb la data d'ara
        setCookie(NOM_COOKIE, new Date().toLocaleString('ca-ES'), DIES_COOKIE);
    }
});
</script>
