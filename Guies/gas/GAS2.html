Bloc II — Llegir la Base de Dades

La primera habilitat fonamental de GAS és llegir dades d'un Spreadsheet. Aquí aprendrem el patró bàsic amb SpreadsheetApp i construirem les dues funcions de lectura que usaran tots els altres scripts.

2.1 El patró bàsic de lectura

SpreadsheetApp.getActiveSpreadsheet()
→
.getSheetByName()"INSCRITS"
→
.getDataRange()tot el rang de dades
→
.getValues()array 2D de valors
Dades.gs — llegirConfig()
// ─────────────────────────────────────────────────────────────
// llegirConfig()
// Llegeix la fulla CONFIG i retorna un objecte clau → valor.
// Ús:  const cfg = llegirConfig();
//      Logger.log(cfg["NOM_JORNADES"]);
// ─────────────────────────────────────────────────────────────
function llegirConfig() {
  const ss    = SpreadsheetApp.getActiveSpreadsheet();
  const fulla = ss.getSheetByName(FULL_CONFIG);

  if (!fulla) {
    Logger.log("ERROR: No s'ha trobat la fulla CONFIG");
    return {};
  }

  const dades = fulla.getDataRange().getValues();
  const cfg   = {};

  for (let i = 1; i < dades.length; i++) { // i=1 salta capçalera
    const clau  = dades[i][0]; // col A → nom del paràmetre
    const valor = dades[i][1]; // col B → valor
    if (clau) cfg[clau] = valor;
  }

  Logger.log("Config llegida: " + Object.keys(cfg).length + " paràmetres");
  return cfg;
}
Dades.gs — llegirInscrits()
// ─────────────────────────────────────────────────────────────
// llegirInscrits()
// Retorna un array d'objectes, un per participant.
// Ús:  const llista = llegirInscrits();
//      llista.forEach(p => Logger.log(p.nom + " → " + p.email));
// ─────────────────────────────────────────────────────────────
function llegirInscrits() {
  const fulla = SpreadsheetApp.getActiveSpreadsheet()
                              .getSheetByName(FULL_INSCRITS);
  const dades = fulla.getDataRange().getValues();
  const inscrits = [];

  for (let i = 1; i < dades.length; i++) {
    const fila = dades[i];
    if (!fila[1]) continue; // salta files buides

    inscrits.push({
      fila:       i + 1,          // número de fila real al Sheets (per escriure-hi)
      timestamp:  fila[0],
      nom:        fila[1],
      centre:     fila[2],
      email:      fila[3],
      telefon:    fila[4],
      activM1:    fila[5],
      activM2:    fila[6],
      activT:     fila[7],
      volSala:    fila[8],
      comentaris: fila[9],
      confirmat:  fila[10]
    });
  }

  Logger.log("Total inscrits llegits: " + inscrits.length);
  return inscrits;
}
💡 Per què retornem objectes i no l'array raw? Accedir a p.email és molt més llegible i menys propens a errors que fila[3]. Si algun dia canvies l'ordre de les columnes, només has d'actualitzar llegirInscrits(), no tots els scripts.

Bloc III — onOpen(): El Menú Personalitzat

La funció onOpen() és un trigger simple que GAS executa automàticament cada cop que s'obre el Spreadsheet. L'usarem per afegir un menú personalitzat "🎓 Jornades" amb accés directe a totes les funcions del sistema.

🧠 Analogia: onOpen() és com el porter d'un hotel que, en arribar, t'obre la porta, et dona la benvinguda i t'indica on és la recepció. S'executa sol, sense que ningú li digui res.

3.1 La funció onOpen() completa

Codi.gs — onOpen() amb menú complet
// ─────────────────────────────────────────────────────────────
// onOpen()
// S'executa automàticament quan s'obre el Spreadsheet.
// Afegeix el menú personalitzat "🎓 Jornades" a la barra de menús.
// ─────────────────────────────────────────────────────────────
function onOpen() {
  const ui  = SpreadsheetApp.getUi();
  const cfg = llegirConfig();

  // Comprova si el sistema està actiu
  const actiu = cfg["ACTIU"] === true || cfg["ACTIU"] === "TRUE";
  const nomJornades = cfg["NOM_JORNADES"] || "Jornades";

  ui.createMenu(`🎓 ${nomJornades}`)

    // ── INSCRIPCIONS ─────────────────────────────────────────
    .addItem("👥 Actualitzar aforaments",        "actualitzarAforaments")
    .addItem("📧 Enviar confirmacions pendents",  "enviarConfirmacionsPendents")
    .addItem("📈 Generar full de Resum",          "generarResum")
    .addSeparator()

    // ── SALA ESPECIAL ────────────────────────────────────────
    .addItem("🚪 Generar agenda Sala Especial",   "generarAgendaSala")
    .addSeparator()

    // ── INFORMES ─────────────────────────────────────────────
    .addSubMenu(
      ui.createMenu("📊 Informes")
        .addItem("Informe d'ocupació per franja",   "informeOcupacio")
        .addItem("Llista per activitat",           "llistaPerActivitat")
        .addItem("Participants Sala Especial",      "llistaSalaEspecial")
    )
    .addSeparator()

    // ── WEB APP ───────────────────────────────────────────────
    .addItem("🌐 Veure URL de la Web App",       "mostrarUrlWebApp")
    .addSeparator()

    // ── SISTEMA ───────────────────────────────────────────────
    .addItem(
      actiu ? "🟢 Sistema ACTIU — clic per desactivar"
            : "🔴 Sistema INACTIU — clic per activar",
      "toggleSistema"
    )
    .addToUi();

  Logger.log("Menú creat per a: " + nomJornades);
}

// Activa o desactiva tot el sistema canviant CONFIG[ACTIU]
function toggleSistema() {
  const fulla = SpreadsheetApp.getActiveSpreadsheet()
    .getSheetByName(FULL_CONFIG);
  const dades = fulla.getDataRange().getValues();

  for (let i = 1; i < dades.length; i++) {
    if (dades[i][0] === "ACTIU") {
      const actual  = dades[i][1];
      const nouValor = (actual === true || actual === "TRUE") ? false : true;
      fulla.getRange(i + 1, 2).setValue(nouValor);
      SpreadsheetApp.getUi().alert(
        "Sistema " + (nouValor ? "ACTIVAT 🟢" : "DESACTIVAT 🔴") +
        "\n\nRe-obre el full per actualitzar el menú."
      );
      return;
    }
  }
}

// Mostra la URL de la Web App publicada en un diàleg
function mostrarUrlWebApp() {
  const cfg = llegirConfig();
  const url = cfg["URL_WEBAPP"] || "Encara no s'ha publicat la Web App.\nFes-ho des del Bloc VII.";
  SpreadsheetApp.getUi().alert("🌐 URL de la Web App", url,
    SpreadsheetApp.getUi().ButtonSet.OK);
}
⚠️ onOpen() no es pot executar manualment amb ▶ — és un trigger automàtic. Per provar-lo: desa el codi i tanca i re-obre el Spreadsheet. Veuràs el menú "🎓 Jornades" a la barra de menús.

Bloc IV — Escriure a Sheets: Aforaments i Resum

Ara aprenem a escriure valors calculats de tornada al Spreadsheet. Les dues funcions principals d'aquesta secció actualitzen l'ocupació de cada activitat i generen el full de Resum per als organitzadors.

4.1 actualitzarAforaments()

Dades.gs — actualitzarAforaments()
function actualitzarAforaments() {
  const cfg = llegirConfig();
  if (cfg["ACTIU"] === false || cfg["ACTIU"] === "FALSE") {
    Logger.log("Sistema inactiu. Operació cancel·lada."); return;
  }

  const ss         = SpreadsheetApp.getActiveSpreadsheet();
  const fullaIns   = ss.getSheetByName(FULL_INSCRITS);
  const fullaAct   = ss.getSheetByName(FULL_ACTIVITATS);
  const inscrits   = fullaIns.getDataRange().getValues();
  const activitats = fullaAct.getDataRange().getValues();

  // 1. Construeix un comptador per activitat
  //    Format: { "M1-A": 5, "M1-B": 3, "T-D": 7 ... }
  const comptador = {};
  for (let i = 1; i < inscrits.length; i++) {
    const f = inscrits[i];
    if (!f[1]) continue;

    // Les opcions del formulari arriben com "A", "B"... — construïm l'ID
    ["M1-"+f[5], "M2-"+f[6], "T-"+f[7]].forEach(id => {
      if (id && id.length > 3) comptador[id] = (comptador[id] || 0) + 1;
    });
  }

  // 2. Escriu els resultats a les columnes H, I, J d'ACTIVITATS
  for (let i = 1; i < activitats.length; i++) {
    const id         = activitats[i][0];  // col A: M1-A, M2-B...
    const aforMax    = activitats[i][6];  // col G: Aforament_Max
    const inscritsN  = comptador[id] || 0;
    const disponibles = aforMax - inscritsN;
    const estat      = inscritsN >= aforMax ? "Completa"
                       : inscritsN > 0     ? "Oberta"
                       : "Oberta";

    const fila = i + 1;
    fullaAct.getRange(fila, 8).setValue(inscritsN);   // H
    fullaAct.getRange(fila, 9).setValue(disponibles);  // I
    fullaAct.getRange(fila, 10).setValue(estat);       // J
  }

  Logger.log("✅ Aforaments actualitzats per a " + (activitats.length-1) + " activitats");
  SpreadsheetApp.getUi().alert("✅ Aforaments actualitzats correctament!");
}

4.2 generarResum()

Dades.gs — generarResum()
function generarResum() {
  const ss       = SpreadsheetApp.getActiveSpreadsheet();
  const cfg      = llegirConfig();
  const inscrits = llegirInscrits();
  const actData  = ss.getSheetByName(FULL_ACTIVITATS).getDataRange().getValues();
  const fullaRes = ss.getSheetByName(FULL_RESUM);

  fullaRes.clearContents();

  // ── Capçalera ────────────────────────────────────────────
  const ara = new Date().toLocaleString("ca-ES");
  fullaRes.getRange("A1").setValue("📊 Resum " + cfg["NOM_JORNADES"]);
  fullaRes.getRange("A2").setValue("Actualitzat: " + ara);

  // ── Totals globals ────────────────────────────────────────
  const totalInscrits  = inscrits.length;
  const totalConfirmats = inscrits.filter(p => p.confirmat === true).length;
  const volSala = inscrits.filter(p => p.volSala === "Sí").length;

  fullaRes.getRange("A4:C4").setValues([["Total inscrits", "Confirmats", "Volen Sala"]]);
  fullaRes.getRange("A5:C5").setValues([[totalInscrits, totalConfirmats, volSala]]);

  // ── Taula d'activitats ────────────────────────────────────
  fullaRes.getRange("A7:E7").setValues([[
    "Activitat", "Títol", "Inscrits", "Places lliures", "Estat"
  ]]);

  for (let i = 1; i < actData.length; i++) {
    const fila = actData[i];
    fullaRes.getRange(7 + i, 1, 1, 5).setValues([[
      fila[0], fila[3], fila[7] || 0, fila[8] || fila[6], fila[9] || "Oberta"
    ]]);
  }

  Logger.log("✅ Resum generat amb " + totalInscrits + " participants");
  SpreadsheetApp.getUi().alert("✅ Full RESUM actualitzat!");
}

Bloc V — GmailApp: Confirmació Automàtica

La funció enviarConfirmacionsPendents() recorre tots els inscrits i envia un correu personalitzat als que encara no han rebut la confirmació (Confirmacio_Enviada = FALSE). Marca la columna K amb TRUE per evitar duplicats.

Correus.gs — enviarConfirmacionsPendents()
function enviarConfirmacionsPendents() {
  const cfg = llegirConfig();
  if (cfg["ACTIU"] === false || cfg["ACTIU"] === "FALSE") return;

  const fulla    = SpreadsheetApp.getActiveSpreadsheet().getSheetByName(FULL_INSCRITS);
  const inscrits = llegirInscrits();
  let enviats = 0;

  const activitats = SpreadsheetApp.getActiveSpreadsheet()
    .getSheetByName(FULL_ACTIVITATS).getDataRange().getValues();

  // Diccionari ID → Títol de l'activitat
  const titols = {};
  activitats.slice(1).forEach(a => { titols[a[0]] = a[3]; });

  inscrits.forEach(p => {
    if (p.confirmat === true || p.confirmat === "TRUE") return; // ja enviat
    if (!p.email) return; // sense email

    try {
      const assumpte = `✅ Confirmació inscripció – ${cfg["NOM_JORNADES"]}`;
      const cos = construirCorreu(p, titols, cfg);

      GmailApp.sendEmail(p.email, assumpte, cos, {
        name:    cfg["NOM_JORNADES"],
        replyTo: cfg["EMAIL_ORGANITZACIO"],
        htmlBody: cos.replace(/\n/g, "<br>")
      });

      // Marca la col K com a TRUE
      fulla.getRange(p.fila, 11).setValue(true);
      enviats++;
      Logger.log("✉️ Enviat a: " + p.email);

    } catch (err) {
      Logger.log("ERROR enviant a " + p.email + ": " + err.message);
    }
  });

  Logger.log(`✅ Total enviats: ${enviats}`);
  SpreadsheetApp.getUi().alert(`✅ S'han enviat ${enviats} confirmacions!`);
}

// ── Construeix el cos del correu de confirmació ──────────────
function construirCorreu(p, titols, cfg) {
  const tM1 = titols["M1-"+p.activM1] || p.activM1;
  const tM2 = titols["M2-"+p.activM2] || p.activM2;
  const tT  = titols["T-"+p.activT]   || p.activT;

  return `Benvolgut/da ${p.nom},

Confirme la teva inscripció a les ${cfg["NOM_JORNADES"]}.

📅 Data: ${cfg["DATA_JORNADES"]}
📍 Lloc: ${cfg["LLOC"]}

Les teves activitats seleccionades:

  🌅 Matí 1  (9:30–11:30): ${tM1}
  ☀️ Matí 2  (12:00–14:00): ${tM2}
  🌆 Tarda   (16:00–18:00): ${tT}
  🚪 Sala Especial: ${p.volSala}

${p.comentaris ? "📝 Les teves observacions han estat registrades." : ""}

Ens veiem el ${cfg["DATA_JORNADES"]}!

Equip organitzador
${cfg["NOM_JORNADES"]}
${cfg["EMAIL_ORGANITZACIO"]}`;
}
⚠️ Quota de Gmail: GmailApp.sendEmail() té un límit de 100 correus/dia amb comptes gratuïts de Google i 1.500/dia amb Google Workspace. Per a jornades amb molts participants, considera enviar en lots o usar un servei de correu professional.

BLOC VI

Bloc VI — Triggers: Automatització Programada

Els triggers (disparadors) permeten que GAS executi funcions automàticament sense que ningú premi cap botó. N'hi ha de dos tipus principals: els que s'activen per un esdeveniment (obrir el full, enviar un formulari) i els que s'activen per temps (cada hora, cada dia a les 8h...).

6.1 Tipus de triggers

TipusQuan s'activaFunció del nostre sistema
onOpenQuan s'obre el SpreadsheetCrear el menú personalitzat
onFormSubmitQuan s'envia el FormulariEnviar confirmació immediatament
Temps · cada horaCada hora exactaActualitzar aforaments en temps real
Temps · diariCada dia a una hora fixadaEnviar informe diari als organitzadors
onEditQuan s'edita una cel·laDetectar canvis d'estat a SALA_ESPECIAL

6.2 Trigger onFormSubmit: confirmació immediata

Aquest és el trigger més potent per al nostre sistema: cada cop que algú envia el formulari, GAS envia automàticament el correu de confirmació en qüestió de segons.

Correus.gs — onFormSubmit() + configuració del trigger
// ─────────────────────────────────────────────────────────────
// onFormSubmit(e)
// S'executa automàticament quan el formulari rep una resposta.
// El paràmetre 'e' conté les dades de la nova fila.
// ─────────────────────────────────────────────────────────────
function onFormSubmit(e) {
  const cfg = llegirConfig();
  if (cfg["ACTIU"] === false || cfg["ACTIU"] === "FALSE") return;

  try {
    // e.values conté els valors de la nova fila enviada
    const valors = e.values;

    const nouInscrit = {
      fila:     e.range.getRow(),
      nom:      valors[1],
      centre:   valors[2],
      email:    valors[3],
      telefon:  valors[4],
      activM1:  valors[5],
      activM2:  valors[6],
      activT:   valors[7],
      volSala:  valors[8],
      comentaris: valors[9],
      confirmat: false
    };

    // Obtenim els títols de les activitats per al correu
    const actData = SpreadsheetApp.getActiveSpreadsheet()
      .getSheetByName(FULL_ACTIVITATS).getDataRange().getValues();
    const titols = {};
    actData.slice(1).forEach(a => { titols[a[0]] = a[3]; });

    // Envia el correu
    const assumpte = `✅ Confirmació – ${cfg["NOM_JORNADES"]}`;
    const cos = construirCorreu(nouInscrit, titols, cfg);
    GmailApp.sendEmail(nouInscrit.email, assumpte, cos, {
      name: cfg["NOM_JORNADES"], replyTo: cfg["EMAIL_ORGANITZACIO"]
    });

    // Marca col K com TRUE
    SpreadsheetApp.getActiveSpreadsheet()
      .getSheetByName(FULL_INSCRITS)
      .getRange(nouInscrit.fila, 11).setValue(true);

    Logger.log("✉️ Confirmació enviada a: " + nouInscrit.email);

  } catch (err) {
    Logger.log("ERROR onFormSubmit: " + err.message);
    // Notifica l'organitzador de l'error
    GmailApp.sendEmail(cfg["EMAIL_NOTIFICACIONS"],
      "⚠️ Error al sistema de confirmacions",
      "Error: " + err.message);
  }
}

// ─────────────────────────────────────────────────────────────
// configurarTriggers()
// Executa aquesta funció UNA SOLA VEGADA per instal·lar els triggers.
// ─────────────────────────────────────────────────────────────
function configurarTriggers() {
  const ss = SpreadsheetApp.getActiveSpreadsheet();

  // Elimina triggers existents per evitar duplicats
  ScriptApp.getProjectTriggers().forEach(t => ScriptApp.deleteTrigger(t));

  // 1. Trigger onFormSubmit → enviament immediat de confirmació
  ScriptApp.newTrigger("onFormSubmit")
    .forSpreadsheet(ss)
    .onFormSubmit()
    .create();

  // 2. Trigger diari → informe de les 8:00h
  ScriptApp.newTrigger("enviarInformeDiari")
    .timeBased()
    .everyDays(1)
    .atHour(8)
    .create();

  // 3. Trigger horari → actualitzar aforaments cada hora
  ScriptApp.newTrigger("actualitzarAforaments")
    .timeBased()
    .everyHours(1)
    .create();

  Logger.log("✅ Triggers configurats correctament");
  SpreadsheetApp.getUi().alert("✅ 3 triggers instal·lats correctament!");
}

7.1 La funció doGet() completa amb rutes

WebApp.gs — doGet() amb router i endpoints
// ─────────────────────────────────────────────────────────────
// doGet(e)
// Punt d'entrada de la Web App. Gestiona les rutes (endpoints):
//
//  /exec                      → info general de les Jornades
//  /exec?ruta=activitats      → totes les activitats amb aforament
//  /exec?ruta=activitat&id=M1-A → detall d'una activitat
//  /exec?ruta=sala            → agenda completa de la Sala Especial
//  /exec?ruta=sala&hora=10:00 → disponibilitat d'una franja
//  /exec?ruta=resum           → resum global d'ocupació
// ─────────────────────────────────────────────────────────────
function doGet(e) {
  const ruta  = e.parameter.ruta  || "";
  const id    = e.parameter.id    || "";
  const hora  = e.parameter.hora  || "";

  let resposta;

  try {
    switch (ruta) {
      case "activitats":
        resposta = apiActivitats();
        break;
      case "activitat":
        resposta = apiActivitat(id);
        break;
      case "sala":
        resposta = hora ? apiSalaHora(hora) : apiSala();
        break;
      case "resum":
        resposta = apiResum();
        break;
      default:
        resposta = apiInfo();
    }
  } catch (err) {
    resposta = { error: true, missatge: err.message };
  }

  return ContentService
    .createTextOutput(JSON.stringify(resposta))
    .setMimeType(ContentService.MimeType.JSON);
}

// ── apiInfo(): informació general ─────────────────────────────
function apiInfo() {
  const cfg = llegirConfig();
  return {
    jornades: cfg["NOM_JORNADES"],
    data:     cfg["DATA_JORNADES"],
    lloc:     cfg["LLOC"],
    endpoints: [
      "?ruta=activitats",
      "?ruta=activitat&id=M1-A",
      "?ruta=sala",
      "?ruta=sala&hora=10:00",
      "?ruta=resum"
    ]
  };
}

// ── apiActivitats(): totes les activitats ─────────────────────
function apiActivitats() {
  const dades = SpreadsheetApp.getActiveSpreadsheet()
    .getSheetByName(FULL_ACTIVITATS).getDataRange().getValues();

  const activitats = dades.slice(1).map(f => ({
    id:          f[0],
    franja:      f[1],
    opcio:       f[2],
    titol:       f[3],
    ponent:      f[4],
    aula:        f[5],
    aforamentMax: f[6],
    inscrits:    f[7] || 0,
    disponibles: f[8] !== "" ? f[8] : f[6],
    estat:       f[9] || "Oberta"
  }));

  return { ok: true, total: activitats.length, activitats };
}

// ── apiActivitat(id): una activitat concreta ──────────────────
function apiActivitat(id) {
  if (!id) return { error: true, missatge: "Cal especificar ?id=M1-A" };
  const tot = apiActivitats();
  const act = tot.activitats.find(a => a.id === id);
  return act
    ? { ok: true, activitat: act }
    : { error: true, missatge: "Activitat no trobada: " + id };
}

// ── apiSala(): agenda completa de la sala ─────────────────────
function apiSala() {
  const dades = SpreadsheetApp.getActiveSpreadsheet()
    .getSheetByName(FULL_SALA).getDataRange().getValues();

  const franges = dades.slice(1).map(f => ({
    horaInici:  f[0],
    horaFi:     f[1],
    estat:      f[5] || "DISPONIBLE",
    responsable: f[2] || null,
    activitat:  f[3] || null
  }));

  const disponibles = franges.filter(f => f.estat === "DISPONIBLE").length;
  return { ok: true, totalFranges: franges.length, disponibles, franges };
}

// ── apiSalaHora(hora): disponibilitat d'una franja ────────────
function apiSalaHora(hora) {
  const sala = apiSala();
  const franja = sala.franges.find(f => f.horaInici === hora);
  return franja
    ? { ok: true, franja }
    : { error: true, missatge: "Franja no trobada: " + hora };
}

// ── apiResum(): estadístiques globals ─────────────────────────
function apiResum() {
  const inscrits   = llegirInscrits();
  const activitats = apiActivitats().activitats;
  const cfg        = llegirConfig();

  const totalPlaces = activitats.reduce((s, a) => s + a.aforamentMax, 0);
  const pctOcupacio = totalPlaces > 0
    ? Math.round(inscrits.length / totalPlaces * 100)
    : 0;

  return {
    ok: true,
    jornades:    cfg["NOM_JORNADES"],
    totalInscrits: inscrits.length,
    totalPlaces,
    pctOcupacio:  pctOcupacio + "%",
    activitatsCompletes: activitats.filter(a => a.estat === "Completa").length,
    actualitzat: new Date().toISOString()
  };
}
